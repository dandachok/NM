\subsection*{Метод Ньютона}

\begin{lstlisting}[language=C++]
double NewtonMethod (double x0, double eps, double (*f)(double), double(*difF)(double)) {
    double x = x0;
    double next_x = x0;
    do {
        x = next_x;
        next_x = x - f(x)/difF(x);
    } while (std::abs(next_x - x) > eps);

    return next_x;
}
\end{lstlisting}
